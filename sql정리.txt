SELECT * FROM 테이블명 WHERE 조건

SELECT * FROM 테이블명 WHERE 조건 AND 조건		~~이고
SELECT * FROM 테이블명 WHERE 조건 OR 조건		~~이거나

연산자

NOT		SELECT * FROM EMP WHERE NOT SAL = 3000;		-> EMP테이블에서 SAL이 3000이 아닌
IN		SELECT * FROM EMP WHERE JOB IN ('MANAGER', 'SALESMAN', 'CLERK');	-> EMP테이블에서 MANAGER, SALESMAN, CLERK를 선택
NOT IN		-> IN과 반대

BETWEEN A AND B 		SELECT * FROM EMP WHERE SAL >= 2000 AND SAL <= 3000;	-> EMP테이블에서 SAL이 2000이상 3000이하인
					->	SELECT * FROM EMP WHERE SAL BETWEEN 2000 AND 3000;
					
					->	SELECT * FROM EMP WHERE SAL NOT BETWEEN 2000 AND 3000; 위의 값과 반대
					
LIKE : 문자열 검색		WHERE 컬럼명 LIKE 'S%';		-> 컬럼에서 S로 시작하는 것을 골라라
					WHERE 컬럼명 LIKE '_S';		-> 컬럼에서 S 앞에 한 글자 있는 것을 골라라
					
					_ 어떤 값이든 상관없이 한 개의 문자 테이터를 의미
					% 길이와 상관없이(문자가 없는 경우도 포함)모든 문자 데이터를 의미

IS NULL : NULL을 확인		SELECT * FROM EMP WHERE COMM IS NULL;		-> EMP테이블의 COMM컬럼에서 NULL인 것을 찾아라
IS NOT NULL : IS NULL과 반대

UNION : 연결된 SELECT문의 결과값을 합집합으로 묶어준다. 중복 값은 제거
UNION ALL : UNION과 같지만 중복 값이 나옴
MINUS : 차집합, 1번 테이블에서 2번 테이블을 뺀다
INTERSECT : 교집합, 1번 테이블과 2번 테이블에서 겹치는 값만 출력
